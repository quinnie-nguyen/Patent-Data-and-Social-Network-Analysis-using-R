descriptive_stats <- function(thu_central, QUALITY_EPO) {
  central <- QUALITY_EPO[QUALITY_EPO$appln_id %in% thu_central$Appln_id, c("appln_id","patent_scope","fwd_cits7")]
  names(central) <- c("Appln_id", "Patent_Scope", "Forward_Citations")
  central <- merge(thu_central, central, by = "Appln_id", all = TRUE )
  central <- unique(central[,c("Appln_id","Prio_Year", "Patent_Scope", "Forward_Citations", "open_or_close")])
  central$Count <- 1
  
  central_pat <- aggregate(central[,c("Patent_Scope", "Forward_Citations", "Count", "open_or_close")], by=list(central$Prio_Year), sum)
  central_pat$AVG_Scope <- round(central_pat$Patent_Scope/central_pat$Count,4)
  central_pat$AVG_FWDCITS <- round(central_pat$Forward_Citations/central_pat$Count, 4)
  names(central_pat)[1] <- c("Period")
  central_pat$Period<- c("1991-1995","1996-2000","2001-2005","2006-2010")
  central_pat$percent_open <- round(central_pat$open_or_close*100/central_pat$Count, 4)
  return(central_pat)
}

Forward citations is the citations received by a patent can be used to assess the technological impact of inventions their cross-technology, geographical impact. The technological impact of inventions can indicate the economic importance of patents. The value of a patent and the number and quality of its forward citations have repeatedly been found to becorrelated. Therefore forward citation is a good indicator to study about technology capabilies.

In our analyse result,the average citations received by patents registered by North in the second and third periods were significantly higher than patents of other sub-regions. It could be explained by the fact that the number of patents in the North was too low so that individual patents had a large impact on the overall view. There were two patents in the north by the Max-Planck-Society on RNA which are highly cited, which could be considered as outliers. Generally, the average citations of patents registered by four sub-regions are smaller than 2, except for North Thuringia in 1996-2000 and 2001-2005. It seems to be difficult to state which sub-region has the most valuable patent during the whole period based on the average measure.

IPC_share_def <- function(CENTRAL_ipc, names_list = c("IPC3", "c1","c2","c3", "c4")){
  ###IPC3_share in every period
      IPC3_share_central <- aggregate(CENTRAL_ipc$Appln_id, by=list(CENTRAL_ipc$Prio_Year, CENTRAL_ipc$ipc3), length)      
      total_ipc <- aggregate(IPC3_share_central$x, by=list(IPC3_share_central$Group.1), sum)
      IPC3_share_central <- merge(IPC3_share_central,total_ipc, all=T, by="Group.1") 
      IPC3_share_central$share <- IPC3_share_central$x.x/IPC3_share_central$x.y
      names(IPC3_share_central) <- names_list
  return(IPC3_share_central)
}

tech_distance_def <- function(IPC3_1){
    IPC3_1[is.na(IPC3_1)] <- 0
    names(IPC3_1) <- c("IPC3", "Central", "East", "South", "North")
    IPC3_1$Diff_CE <- (IPC3_1$Central -IPC3_1$East)**2
    IPC3_1$Diff_CS <- (IPC3_1$Central -IPC3_1$South)**2
    IPC3_1$Diff_CN <- (IPC3_1$Central -IPC3_1$North)**2
    IPC3_1$Diff_ES <- (IPC3_1$East -IPC3_1$South)**2
    IPC3_1$Diff_EN <- (IPC3_1$East -IPC3_1$North)**2
    IPC3_1$Diff_SN <- (IPC3_1$South -IPC3_1$North)**2
    CE <- round(sqrt(sum(IPC3_1$Diff_CE)),3)
    CS <- round(sqrt(sum(IPC3_1$Diff_CS)),3)
    CN <- round(sqrt(sum(IPC3_1$Diff_CN)),3)
    ES <- round(sqrt(sum(IPC3_1$Diff_ES)),3)
    EN <- round(sqrt(sum(IPC3_1$Diff_EN)),3)
    SN <- round(sqrt(sum(IPC3_1$Diff_SN)),3)
    return(c(CE, CS, CN, ES, EN, SN))
}
